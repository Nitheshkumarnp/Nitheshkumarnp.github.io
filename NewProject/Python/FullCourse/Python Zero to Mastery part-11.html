	from selenium import webdriver
	chrome_browser = webdriver.Chrome('./chromedriver')
	print(chrome_browser)		// it opens chrome app

	chrome_browser.maximize_window()	// it maximize the chrome window
	chrome_browser.window()		// it open another chrome window with normal size

Selenium basics2:
	from selenium import webdriver
	chrome_browser = webdriver.Chrome('./chromedriver')
	chrome_browser.maximize_window()
	chrome_browser.get('https://www.seleniumeasy.com/test/basic')
	assert 'Show message' in chrome_browser.page_source
	element = chrome_browser.find_element_by_id('user-message')
	element.clear()
	element.send_keys('nikprime')
	element2 = chrome_browser.find_element_by_class_name('btn-default')
	element2.click()

	// send_keys send the nikprime to input tag in chrome
	// click will automatically click the button show message and 
	// show the nikprime string.

	output = chrome_browser.find_element_by_id('display')
	assert 'nikprime' in output.text
	
	// it checks the nikprime is found in display id.

	chrome_browser.find_element_by_css_select('#get-input > .btn')

Selenium basics3:
	whenever we run the above code, it always open a new chrome window
		it should not be good and quite.
	so use .close(), it automatically close the window after run the code.
	
	chrome_browser.close()
	chrome_browser.quit()		// it quit all the windows of chrome.
