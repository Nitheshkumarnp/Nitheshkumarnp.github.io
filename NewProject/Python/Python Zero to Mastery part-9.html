	import requests
	from bs4 import BeautifulSoup
	res = requests.get('https://news.ycombinator.com/news')
	print(res.text)			// it shows the html file as text format
	BeautifulSoup(res.text, 'html.parser')
	print(soup)			// it shows some clean html file as html
	print(soup.body)		// it shows the body content only
	print(soup.body.contents)	// it shows the contents only
	print(soup.find_all('div'))	// it shows all the divs
	print(soup.title)		// it shows the title of page
	print(soup.select('a'))		// it selects all a tags
	print(soup.select('.score'))	// it shows all spam class having score
	print(soup.select('.storylink')[0])	// it shows first link having storylink as class
	links = soup.select('.storylink')
	votes = soup.select('.score')
	print(votes[0])			// it shows the votes of soup
	
	def create_custom_hn(links, votes):
